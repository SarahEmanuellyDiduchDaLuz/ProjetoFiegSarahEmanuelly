Como Funciona

A extensão coleta o conteúdo textual da página e envia para análise inteligente.

content.js

const texto = document.body.innerText; chrome.runtime.sendMessage({ action: "analisarTexto", payload: texto });

background.js

fetch("https://sua-api.com/analisar", { method: "POST", body: JSON.stringify({ texto: msg.payload }) }) .then(r => r.json()) .then(data => { console.log(data); });
